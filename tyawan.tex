\documentclass{jarticle}

\begin{document}
	\title{茶碗蒸しのレシピ(1人分)}
	\maketitle

	\begin{itembox}[l]{用意するもの}
		\begin{enumerate}
			\item 卵1個
			\item 出汁の素
			\item 塩
		\end{enumerate}
	\end{itembox}

	\begin{enumerate}
		\item 卵に水を注ぎ、160mLにする。
		\item 泡立て器で撹拌する。
		\item 出汁と塩を小匙1いれて、撹拌する。
		\item 湯呑みやビーカーなどの容器に注ぎ、泡を匙などで取る。
		\item 強火で2分蒸す。
		\item 弱火で8分蒸す。
	\end{enumerate}
